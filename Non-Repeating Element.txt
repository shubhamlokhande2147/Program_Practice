Non-Repeating Element

class Check{
    public int firstNonRepeating(int arr[], int n) 
    { 
        for (int i = 0; i < n; i++) {
            int j;
            // Checking if ith element is present in array
            for (j = 0; j < n; j++)
                if (i != j && arr[i] == arr[j])
                    break;
            // if ith element is not present in array
            // except at ith index then return element
            if (j == n)
                return arr[i];
        }
        return 0;
    }    
}
